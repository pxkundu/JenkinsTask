1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  export INGRESS_HOST=$(kubectl -n {{ .Release.Namespace }} get ingress {{ .Release.Name }}-todo -o jsonpath="{.spec.rules[0].host}")
  echo "Visit http://$INGRESS_HOST to access your application"
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app={{ .Release.Name }}-frontend" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8080/ to access your frontend"
  kubectl port-forward $POD_NAME 8080:{{ .Values.service.frontend.port }} --namespace {{ .Release.Namespace }}
  echo "Backend available internally at {{ .Release.Name }}-backend:{{ .Values.service.backend.port }}"
{{- end }}
